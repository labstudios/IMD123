START

fileName:String = "movies.csv"

titleId:Integer = 0

directorId:Integer = 1

durationId:Integer = 2

movieFile:File

currentLine:String

movieData:Array


readFile fileName to movieFile


while movieFile !eof

currentLine = read line of movieFile

movieData = split(",",currentLine)

print "Title: " + movieData[titleId] + "\r\n"

print "Directed by " + movieData[directorId] + "\r\n"

print "Run time: " + movieData[durationId] + "\r\n"

print "-------------------------------" + "\r\n\r\n"


end while

END //It's Over!
